```sql
-- 查看各年级题目数量
SELECT gu.grade, gu.semester, COUNT(q.id) as question_count
FROM tb_grade_unit gu
LEFT JOIN tb_question q ON gu.id = q.grade_unit_id
GROUP BY gu.grade, gu.semester
ORDER BY gu.grade, gu.semester;

-- 查看题目类型分布
SELECT question_type, COUNT(*) as count
FROM tb_question
GROUP BY question_type;
```
